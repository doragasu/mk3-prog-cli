247 static int ParseMemArgument(MemImage *m) {
248  int i;
249  char *addr = NULL;
250  char *len = NULL;
251  char *endPtr;
252 
253  // Set address and length to default values
254  m->len = m->addr = 0;
255 
256  // First argument is name. Find where it ends
257  for (i = 0; i < (MAX_FILELEN + 1) && m->file[i] != '\0' &&
258  m->file[i] != ':'; i++);
259  // Check if end reached without finding end of string
260  if (i == MAX_FILELEN + 1) return 1;
261  if (m->file[i] == '\0') return 0;
262 
263  // End of token marker, search address
264  m->file[i++] = '\0';
265  addr = m->file + i;
266  for (; i < (MAX_FILELEN + 1) && m->file[i] != '\0' && m->file[i] != ':';
267  i++);
268  // Check if end reached without finding end of string
269  if (i == MAX_FILELEN + 1) return 1;
270  // If end of token marker, search length
271  if (m->file[i] == ':') {
272  m->file[i++] = '\0';
273  len = m->file + i;
274  // Verify there's an end of string
275  for (; i < (MAX_FILELEN + 1) && m->file[i] != '\0'; i++);
276  if (m->file[i] != '\0') return 1;
277  }
278  // Convert strings to numbers and return
279  if (addr && *addr) m->addr = strtol(addr, &endPtr, 0);
280  if (m->addr == 0 && addr == endPtr) return 2;
281  if (len && *len) m->len = strtol(len, &endPtr, 0);
282  if (m->len == 0 && len == endPtr) return 3;
283 
284  return 0;
285 }
