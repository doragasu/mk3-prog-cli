54 // NOTE: Previous implementation sent each frame split in 3 steps to avoid
55 // having to copy input data: SOF + len, payload, eof. This caused huge delays
56 // due minimum 1 ms USB latency. Due to this implementation has been changed
57 // to build the complete frame (even if this means copying the data) and send
58 // it at once.
59 int SCFrameSend(struct mpsse_context *mpsse, char *data, uint16_t datalen) {
60  // Frame buffer. Maximum length is the payload length + SOF + EOF + LEN.
61  char frame[SC_MAX_DATALEN + 3];
62  uint16_t rem;
63  uint16_t sent;
64 
65  frame[0] = SC_SOF;
66  // If payload greater than SC_MAX_DATALEN bytes, split data
67  // transmission in SC_MAX_DATALEN byte chuncks.
68  for (sent = 0, rem = datalen; 0 < rem; sent += frame[1], rem -= frame[1]) {
69  // Build frame
70  frame[1] = MIN(rem, SC_MAX_DATALEN);
71  memcpy(frame + 2, data + sent, frame[1]);
72  frame[frame[1] + 2] = SC_EOF;
73  // Send frame
74  if (Start(mpsse)) SCStopErr();
75  if (Write(mpsse, frame, frame[1] + 3)) SCStopErr();
76  if (Stop(mpsse)) SCStopErr();
77  }
78 
79  return SC_OK;
80 }
